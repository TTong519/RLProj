### Task 1: Add `train-rllib` CLI command
**Action:** Append a `train_rllib` Typer command to `cli.py`.

**Arguments:**
| Flag | Type | Default | Maps to |
|---|---|---|---|
| `--scene` | str | `scenes/simple_suturing.json` | `RllibConfig.scene_path` |
| `--algorithm` | str | `PPO` | `RllibConfig.algorithm` |
| `--timesteps` | int | `100000` | `RllibConfig.total_timesteps` |
| `--n-envs` | int | `4` | `RllibConfig.num_env_runners + 1` |
| `--save-dir` | str | `logs/rllib` | `RllibConfig.save_dir` |
| `--checkpoint-freq` | int | `10000` | `RllibConfig.checkpoint_freq` |
| `--lr` | float | `3e-4` | `RllibConfig.lr` |
| `--gamma` | float | `0.99` | `RllibConfig.gamma` |
| `--seed` | int | `42` | `RllibConfig.seed` |
| `--config-file` | str | None | Load JSON config, override with CLI args |

**Pattern:**
```python
@app.command()
def train_rllib(
    scene: str = typer.Option("scenes/simple_suturing.json", "--scene", help="Scene definition file"),
    algorithm: str = typer.Option("PPO", "--algorithm", help="Algorithm (PPO, SAC)"),
    timesteps: int = typer.Option(100000, "--timesteps", "-t", help="Total training timesteps"),
    n_envs: int = typer.Option(4, "--n-envs", help="Number of parallel environments"),
    save_dir: str = typer.Option("logs/rllib", "--save-dir", help="Checkpoint save directory"),
    checkpoint_freq: int = typer.Option(10000, "--checkpoint-freq", help="Checkpoint frequency"),
    lr: float = typer.Option(3e-4, "--lr", help="Learning rate"),
    gamma: float = typer.Option(0.99, "--gamma", help="Discount factor"),
    seed: int = typer.Option(42, "--seed", help="Random seed"),
    config_file: str = typer.Option(None, "--config", help="JSON config file to load"),
) -> None:
    """Train with Ray RLlib (distributed-capable)."""
    try:
        from surg_rl.rl.rllib import RllibConfig, train_rllib as run_train
    except ImportError as e:
        console.print(f"[bold red]Error:[/bold red] {e}")
        console.print('[dim]Install with: pip install "surg-rl[distributed]"[/dim]')
        raise typer.Exit(1)

    # Load from config file if provided
    if config_file:
        with open(config_file) as f:
            data = json.load(f)
        rllib_config = RllibConfig(**data)
    else:
        rllib_config = RllibConfig(
            scene_path=scene,
            algorithm=algorithm.upper(),
            total_timesteps=timesteps,
            num_env_runners=max(0, n_envs - 1),
            save_dir=save_dir,
            checkpoint_freq=checkpoint_freq,
            lr=lr,
            gamma=gamma,
            seed=seed,
        )

    console.print(f"[bold blue]RLlib Training[/bold blue] | algorithm={algorithm} | timesteps={timesteps}")
    console.print(f"[dim]Ray resources will be auto-detected...[/dim]")

    checkpoint = run_train(rllib_config)
    console.print(f"[bold green]✓[/bold green] Final checkpoint: {checkpoint}")
```

**Critical constraints:**
- Catch `ImportError` gracefully and print install instructions (Ray is optional)
- `algorithm.upper()` to normalize "ppo" -> "PPO"
- `num_env_runners = max(0, n_envs - 1)` because RLlib counts env_runners separately from the main process env

**Verification:** `surg-rl train-rllib --timesteps 100 --n-envs 2` runs without crash when ray is installed.

### Task 2: Add `tune` CLI command
**Action:** Append a `tune` Typer command for hyperparameter search.

**Arguments:**
| Flag | Type | Default | Description |
|---|---|---|---|
| `--scene` | str | ... | Base scene (can be overridden by search space) |
| `--algorithm` | str | `PPO` | Base algorithm |
| `--num-samples` | int | `3` | Number of Tune trials |
| `--max-iterations` | int | `10` | Max training iterations per trial |
| `--scenes` | list[str] | None | Categorical scene search space |
| `--lr-min` / `--lr-max` | float | `1e-5` / `1e-3` | LR search range |

**Pattern:**
```python
@app.command()
def tune(
    scene: str = typer.Option("scenes/simple_suturing.json", "--scene"),
    algorithm: str = typer.Option("PPO", "--algorithm"),
    num_samples: int = typer.Option(3, "--num-samples", "-n"),
    max_iterations: int = typer.Option(10, "--max-iterations"),
    scenes: list[str] = typer.Option(None, "--scenes", help="Scene files for categorical search"),
    lr_min: float = typer.Option(1e-5, "--lr-min"),
    lr_max: float = typer.Option(1e-3, "--lr-max"),
) -> None:
    """Run hyperparameter search with Ray Tune + RLlib."""
    try:
        from surg_rl.rl.rllib import (
            RllibConfig,
            build_tune_search_space,
            run_tune_experiment,
        )
    except ImportError as e:
        console.print(f"[bold red]Error:[/bold red] {e}")
        raise typer.Exit(1)

    base_config = RllibConfig(
        scene_path=scene,
        algorithm=algorithm.upper(),
        total_timesteps=10000,  # Short runs for Tune
        num_env_runners=1,
        num_learners=0,
        num_gpus_per_learner=0,
    )

    scene_list = scenes if scenes else [scene]
    param_space = build_tune_search_space(
        base_config,
        scene_paths=scene_list,
        lr_range=(lr_min, lr_max),
    )

    console.print(f"[bold blue]Tune Experiment[/bold blue] | samples={num_samples} | max_iter={max_iterations}")
    results = run_tune_experiment(
        base_config,
        param_space,
        num_samples=num_samples,
        max_training_iterations=max_iterations,
    )

    best = results.get_best_result()
    console.print(f"[bold green]✓[/bold green] Best trial: {best.path}")
    console.print(f"  reward={best.metrics.get('env_runners/episode_return_mean', 'N/A')}")
```

**Verification:** `surg-rl tune --num-samples 2 --max-iterations 2` runs 2 trials without crash.

### Task 4: Add `checkpoint-inspect` CLI command
**Action:** Append `checkpoint_inspect` command for DIST-05 usability.

**Pattern:**
```python
@app.command()
def checkpoint_inspect(
    path: str = typer.Argument(..., help="Checkpoint file or directory"),
    compare_with: str = typer.Option(None, "--compare", help="Path to other checkpoint for comparison"),
) -> None:
    """Inspect RLlib or SB3 checkpoint layer shapes."""
    from pathlib import Path

    path_obj = Path(path)
    is_rllib = path_obj.is_dir()

    if is_rllib:
        try:
            from surg_rl.rl.rllib import inspect_rllib_checkpoint
            info = inspect_rllib_checkpoint(path)
        except ImportError:
            console.print("[bold red]Ray not installed[/bold red]")
            raise typer.Exit(1)
    else:
        from surg_rl.rl.rllib import inspect_sb3_checkpoint
        info = inspect_sb3_checkpoint(path)

    console.print(f"[bold]Format:[/bold] {info['format']}")
    console.print(f"[bold]Algorithm:[/bold] {info.get('algorithm', 'unknown')}")
    console.print("[bold]Layer shapes:[/bold]")
    for name, shape in info.get("layer_shapes", {}).items():
        console.print(f"  {name}: {shape}")

    if compare_with:
        from surg_rl.rl.rllib import compare_checkpoints
        if is_rllib:
            comparison = compare_checkpoints(path, compare_with)
        else:
            comparison = compare_checkpoints(compare_with, path)
        console.print("\n[bold]Comparison:[/bold]")
        console.print(comparison["notes"])
```

**Verification:** `surg-rl checkpoint-inspect logs/rllib/final` shows layer shapes.
