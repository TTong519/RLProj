### Task 1: Create `surg_rl/rl/rllib/tune_integration.py`
**Action:** Implement `build_tune_search_space()` and `run_tune_experiment()` functions.

**`build_tune_search_space()` responsibilities:**
- Accept a base `RllibConfig` and override specific fields with `tune` distribution objects
- Return a `dict` or `AlgorithmConfig` with tune placeholders that `tune.Tuner` can consume
- Support categorical search over: `scene_path`, `simulator_type`, `algorithm`
- Support continuous search over: `lr`, `gamma`, `clip_param`, `ent_coef` (PPO), `tau` (SAC)
- Support nested search over reward weights inside `env_config.reward_config`

**Pattern:**
```python
from __future__ import annotations

from typing import Any

from ray import tune
from ray.rllib.algorithms.ppo import PPOConfig
from ray.rllib.algorithms.sac import SACConfig

from surg_rl.rl.environment import SurgicalEnvConfig
from surg_rl.rl.rewards import RewardConfig

from .config import RllibConfig


def build_tune_search_space(
    base_config: RllibConfig,
    scene_paths: list[str] | None = None,
    simulator_types: list[str] | None = None,
    algorithms: list[str] | None = None,
    lr_range: tuple[float, float] = (1e-5, 1e-3),
    gamma_range: tuple[float, float] = (0.95, 0.999),
    reward_weight_ranges: dict[str, tuple[float, float]] | None = None,
) -> dict[str, Any]:
    """Build a Tune search space from a base RllibConfig.

    Returns a param_space dict suitable for tune.Tuner.
    """
    param_space: dict[str, Any] = {}

    # Categorical overrides
    if scene_paths:
        param_space["scene_path"] = tune.choice(scene_paths)
    if simulator_types:
        param_space["simulator_type"] = tune.choice(simulator_types)
    if algorithms:
        param_space["algorithm"] = tune.choice(algorithms)

    # Continuous overrides
    param_space["lr"] = tune.loguniform(*lr_range)
    param_space["gamma"] = tune.uniform(*gamma_range)

    # Algorithm-specific
    if base_config.algorithm == "PPO":
        param_space["clip_param"] = tune.uniform(0.05, 0.3)
        param_space["entropy_coeff"] = tune.loguniform(1e-4, 0.1)
    elif base_config.algorithm == "SAC":
        param_space["tau"] = tune.uniform(0.001, 0.02)

    # Reward weights nested in env_config
    if reward_weight_ranges:
        param_space["env_config"] = {
            "reward_config": {
                name: tune.uniform(low, high)
                for name, (low, high) in reward_weight_ranges.items()
            }
        }

    return param_space
```

**`run_tune_experiment()` responsibilities:**
- Accept a base `RllibConfig`, search space, and tune config
- Build `tune.Tuner` with `ASHAScheduler` (efficient early stopping)
- Run experiment and return `ResultGrid`
- Save best config to JSON

**Pattern:**
```python
def run_tune_experiment(
    base_config: RllibConfig,
    param_space: dict[str, Any],
    num_samples: int = 3,
    max_training_iterations: int = 10,
    metric: str = "env_runners/episode_return_mean",
    mode: str = "max",
    scheduler: str = "asha",
) -> tune.ResultGrid:
    """Run a Ray Tune experiment over RLlib hyperparameters."""
    from ray import tune
    from ray.tune.schedulers import ASHAScheduler, PopulationBasedTraining

    # Build base RLlib config as param_space foundation
    rllib_base = base_config.build_rllib_config()

    # Merge user param_space into the RLlib config param_space
    # RLlib AlgorithmConfig objects can be used directly as param_space
    # by overriding specific fields with tune objects
    merged_param_space = rllib_base
    for key, value in param_space.items():
        if key == "lr":
            merged_param_space = merged_param_space.training(lr=value)
        elif key == "gamma":
            merged_param_space = merged_param_space.training(gamma=value)
        elif key == "clip_param":
            merged_param_space = merged_param_space.training(clip_param=value)
        elif key == "entropy_coeff":
            merged_param_space = merged_param_space.training(entropy_coeff=value)
        elif key == "tau":
            merged_param_space = merged_param_space.training(tau=value)
        elif key == "env_config":
            # Merge nested env_config carefully
            existing = merged_param_space.get("env_config") or {}
            existing.update(value)
            merged_param_space = merged_param_space.environment(
                base_config.env_name, env_config=existing
            )
        # ... other mappings

    # Scheduler
    if scheduler == "asha":
        sched = ASHAScheduler(
            metric=metric,
            mode=mode,
            max_t=max_training_iterations,
            grace_period=1,
            reduction_factor=2,
        )
    elif scheduler == "pbt":
        sched = PopulationBasedTraining(
            metric=metric,
            mode=mode,
            perturbation_interval=5,
            hyperparam_mutations={
                "lr": tune.loguniform(1e-5, 1e-3),
            },
        )
    else:
        sched = None

    tuner = tune.Tuner(
        base_config.algorithm.lower(),  # "ppo" or "sac"
        param_space=merged_param_space,
        tune_config=tune.TuneConfig(
            metric=metric,
            mode=mode,
            num_samples=num_samples,
            scheduler=sched,
        ),
        run_config=tune.RunConfig(
            name="surg_rl_tune",
            stop={"training_iteration": max_training_iterations},
        ),
    )

    results = tuner.fit()

    # Log best result
    best = results.get_best_result()
    logger.info("Best trial: %s | reward=%.2f", best.path, best.metrics[metric])

    return results
```

**Critical constraint:** The merged param_space approach above is complex. A simpler and more reliable pattern is to use `tune.with_parameters()` or define a trainable function that constructs the RLlib config inside the trial. However, the cleanest RLlib-native pattern is to use the AlgorithmConfig object as param_space and override fields with tune objects.

For simplicity and reliability, use the **trainable function pattern**:
```python
def _trainable_fn(config: dict) -> None:
    """Inner trainable for Tune."""
    from .config import RllibConfig
    from .train import train_rllib

    rllib_cfg = RllibConfig(**config["rllib"])
    # Override with tune-sampled values
    for key, value in config["overrides"].items():
        setattr(rllib_cfg, key, value)

    train_rllib(rllib_cfg)
```

**Pitfall guard:** Don't build the full AlgorithmConfig with tune objects inside — RLlib's config builder may evaluate tune objects prematurely. Use the trainable function pattern or the dict-param-space pattern shown in the RESEARCH.md code examples.

### Task 2: Reward Weight Search Space Serialization
**Action:** `RewardConfig` is a dataclass. When passed as nested dict in `env_config`, RLlib serializes it across actors. Ensure the dict contains only JSON-safe values (floats, not objects).

**Pattern:**
```python
# In build_tune_search_space
reward_ranges = {
    "distance_weight": (0.1, 2.0),
    "success_bonus": (10.0, 100.0),
    "collision_penalty": (-10.0, -1.0),
}
param_space["env_config"] = {
    "reward_config": {
        k: tune.uniform(low, high)
        for k, (low, high) in reward_ranges.items()
    }
}
```

The `env_creator` in `env_wrapper.py` receives this dict and passes it to `SurgicalEnvConfig(**env_config)`. Since `RewardConfig` is a dataclass, `SurgicalEnvConfig` won't auto-convert the nested dict. We need to handle this.

**Fix in env_wrapper.py:**
```python
def make_surgical_env(env_config: dict[str, Any] | None = None) -> SurgicalEnv:
    env_config = env_config or {}
    env_config["render_mode"] = None

    # Handle nested reward_config dict -> RewardConfig dataclass
    if "reward_config" in env_config and isinstance(env_config["reward_config"], dict):
        from surg_rl.rl.rewards import RewardConfig
        env_config["reward_config"] = RewardConfig(**env_config["reward_config"])

    cfg = SurgicalEnvConfig(**env_config)
    return SurgicalEnv(cfg)
```
