```python
from __future__ import annotations

from pathlib import Path
from typing import Any

from surg_rl.utils.logging import get_logger

logger = get_logger(__name__)


def inspect_rllib_checkpoint(checkpoint_dir: str) -> dict[str, Any]:
    """Inspect an RLlib checkpoint and return layer shape metadata.

    Args:
        checkpoint_dir: Path to RLlib checkpoint directory.

    Returns:
        Dict with keys: algorithm, framework, layer_shapes, env_name, total_timesteps.
    """
    checkpoint_dir = Path(checkpoint_dir)
    if not checkpoint_dir.exists():
        raise FileNotFoundError(f"Checkpoint not found: {checkpoint_dir}")

    result: dict[str, Any] = {
        "format": "rllib",
        "path": str(checkpoint_dir),
        "layer_shapes": {},
    }

    # Try to load metadata.json for algorithm info
    metadata_path = checkpoint_dir / "metadata.json"
    if metadata_path.exists():
        import json
        with open(metadata_path) as f:
            meta = json.load(f)
        result["algorithm"] = meta.get("algorithm", "unknown")
        result["env_name"] = meta.get("env_name", "unknown")

    # Try to load RLModule state dict for shape inspection
    rl_module_path = (
        checkpoint_dir
        / "learner_group"
        / "learner"
        / "rl_module"
        / "default_policy"
    )
    if rl_module_path.exists():
        try:
            from ray.rllib.core.rl_module.rl_module import RLModule
            import torch

            rl_module = RLModule.from_checkpoint(str(rl_module_path))
            state = rl_module.get_state()
            for name, param in state.items():
                if isinstance(param, torch.Tensor):
                    result["layer_shapes"][name] = tuple(param.shape)
        except ImportError:
            logger.warning("Ray not installed; cannot inspect RLModule shapes")
        except Exception as e:
            logger.warning("Failed to inspect RLModule: %s", e)

    return result
```

**`inspect_sb3_checkpoint` pattern:**
```python
def inspect_sb3_checkpoint(checkpoint_path: str) -> dict[str, Any]:
    """Inspect an SB3 checkpoint and return layer shape metadata.

    Args:
        checkpoint_path: Path to SB3 .zip checkpoint.

    Returns:
        Dict with keys: algorithm, framework, layer_shapes, policy_type.
    """
    checkpoint_path = Path(checkpoint_path)
    if not checkpoint_path.exists():
        raise FileNotFoundError(f"Checkpoint not found: {checkpoint_path}")

    result: dict[str, Any] = {
        "format": "sb3",
        "path": str(checkpoint_path),
        "layer_shapes": {},
    }

    try:
        from stable_baselines3 import PPO, SAC
        import zipfile
        import torch

        # SB3 .zip contains policy.pth — load it
        with zipfile.ZipFile(checkpoint_path, "r") as z:
            policy_bytes = z.read("policy.pth")

        # Load state dict from bytes
        import io
        state_dict = torch.load(
            io.BytesIO(policy_bytes),
            map_location="cpu",
            weights_only=False,
        )

        for name, param in state_dict.items():
            if hasattr(param, "shape"):
                result["layer_shapes"][name] = tuple(param.shape)

        # Try to detect algorithm from zip contents
        with zipfile.ZipFile(checkpoint_path, "r") as z:
            names = z.namelist()
            if "ppo_policy.pth" in names or "policy.pth" in names:
                result["algorithm"] = "PPO"
            elif "sac_policy.pth" in names:
                result["algorithm"] = "SAC"
            else:
                result["algorithm"] = "unknown"

    except ImportError:
        logger.warning("stable-baselines3 not installed")
    except Exception as e:
        logger.warning("Failed to inspect SB3 checkpoint: %s", e)

    return result
```

**`compare_checkpoints` pattern:**
```python
def compare_checkpoints(rllib_dir: str, sb3_path: str) -> dict[str, Any]:
    """Compare RLlib and SB3 checkpoint layer shapes.

    Returns a dict with:
    - rllib_shapes: dict[str, tuple]
    - sb3_shapes: dict[str, tuple]
    - shared_input_dim: int | None (observation space flat dim)
    - shared_output_dim: int | None (action space flat dim)
    - notes: str (human-readable migration guidance)
    """
    rllib_info = inspect_rllib_checkpoint(rllib_dir)
    sb3_info = inspect_sb3_checkpoint(sb3_path)

    rllib_shapes = rllib_info.get("layer_shapes", {})
    sb3_shapes = sb3_info.get("layer_shapes", {})

    # Heuristic: find input/output dims from first/last layer shapes
    def _infer_io_dims(shapes: dict[str, tuple]) -> tuple[int | None, int | None]:
        if not shapes:
            return None, None
        # Sort by param size to guess order
        sorted_items = sorted(shapes.items(), key=lambda x: x[1][0] if len(x[1]) > 0 else 0)
        input_dim = sorted_items[0][1][0] if sorted_items else None
        output_dim = sorted_items[-1][1][0] if sorted_items else None
        return input_dim, output_dim

    rllib_in, rllib_out = _infer_io_dims(rllib_shapes)
    sb3_in, sb3_out = _infer_io_dims(sb3_shapes)

    notes = (
        "RLlib and SB3 use different internal architectures (RLModule vs MlpExtractor). "
        "Layer names do not match. Weight transfer requires manual mapping of layer shapes.\n\n"
        "Migration steps:\n"
        "1. Inspect both checkpoints with inspect_rllib_checkpoint() and inspect_sb3_checkpoint()\n"
        "2. Match layers by input/output dimensions (observation_dim -> hidden -> action_dim)\n"
        "3. Copy weights tensor-by-tensor using torch.nn.Parameter\n"
        "4. Save as new model in target framework\n\n"
        "Note: Full automatic conversion is not supported due to architecture differences."
    )

    return {
        "rllib_shapes": rllib_shapes,
        "sb3_shapes": sb3_shapes,
        "rllib_input_dim": rllib_in,
        "rllib_output_dim": rllib_out,
        "sb3_input_dim": sb3_in,
        "sb3_output_dim": sb3_out,
        "input_dim_match": rllib_in == sb3_in if (rllib_in and sb3_in) else None,
        "output_dim_match": rllib_out == sb3_out if (rllib_out and sb3_out) else None,
        "notes": notes,
    }
```

**Critical constraints:**
- RLlib checkpoint loading requires `ray` to be installed (RLModule import). Guard with try/except and log warnings.
- SB3 checkpoint loading requires `stable-baselines3` (already a core dependency).
- Both functions must work even when the other framework is missing (e.g., inspect RLlib checkpoint on a machine with only SB3 installed).
- `zipfile` + `io.BytesIO` is the correct way to read `.pth` from inside an SB3 `.zip` without extracting.
