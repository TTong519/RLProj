### Task 1: Add Ray fixture to `tests/conftest.py`
**Action:** Append a pytest fixture that initializes Ray before tests and shuts it down after.

**Pattern:**
```python
@pytest.fixture
def ray_init():
    """Initialize Ray for RLlib tests."""
    import ray
    if not ray.is_initialized():
        ray.init(num_cpus=2, num_gpus=0, ignore_reinit_error=True)
    yield
    if ray.is_initialized():
        ray.shutdown()
```

**Critical constraint:** Use `num_cpus=2, num_gpus=0` for CI/headless environments. Mark tests that need this fixture explicitly.

```python
def test_train_rllib_keyboard_interrupt(ray_init, minimal_config, tmp_path, monkeypatch):
    """Simulate KeyboardInterrupt mid-training."""
    call_count = [0]
    original_train = None

    def interrupting_train(self):
        call_count[0] += 1
        if call_count[0] >= 2:
            raise KeyboardInterrupt()
        return original_train(self)

    from ray.rllib.algorithms.algorithm import Algorithm
    original_train = Algorithm.train
    monkeypatch.setattr(Algorithm, "train", interrupting_train)

    checkpoint = train_rllib(minimal_config)
    assert Path(checkpoint).exists()
    assert "interrupted" in checkpoint
```

```python
import pytest
import json
from pathlib import Path

from surg_rl.rl.rllib.checkpoint_utils import (
    inspect_rllib_checkpoint,
    inspect_sb3_checkpoint,
    compare_checkpoints,
)


def test_inspect_rllib_checkpoint_metadata(tmp_path):
    ckpt = tmp_path / "rllib_ckpt"
    ckpt.mkdir()
    (ckpt / "metadata.json").write_text(json.dumps({"algorithm": "PPO", "env_name": "surg-rl"}))
    info = inspect_rllib_checkpoint(str(ckpt))
    assert info["format"] == "rllib"
    assert info["algorithm"] == "PPO"
    assert info["env_name"] == "surg-rl"


def test_inspect_rllib_without_ray_warns(tmp_path, monkeypatch):
    """Ray not installed: still returns metadata, warns about shapes."""
    monkeypatch.setitem(__import__("sys").modules, "ray", None)
    ckpt = tmp_path / "rllib_ckpt"
    ckpt.mkdir()
    (ckpt / "metadata.json").write_text('{"algorithm": "PPO"}')
    info = inspect_rllib_checkpoint(str(ckpt))
    assert info["format"] == "rllib"
    # layer_shapes will be empty because RLModule can't be imported


def test_inspect_sb3_checkpoint_shapes(tmp_path):
    """Create a mock SB3 .zip with policy.pth."""
    import zipfile, torch, io
    zip_path = tmp_path / "sb3_model.zip"
    state_dict = {"actor.weight": torch.randn(64, 10), "actor.bias": torch.randn(64)}
    policy_bytes = io.BytesIO()
    torch.save(state_dict, policy_bytes)
    policy_bytes.seek(0)

    with zipfile.ZipFile(zip_path, "w") as z:
        z.writestr("policy.pth", policy_bytes.read())

    info = inspect_sb3_checkpoint(str(zip_path))
    assert info["format"] == "sb3"
    assert "actor.weight" in info["layer_shapes"]
    assert info["layer_shapes"]["actor.weight"] == (64, 10)


def test_compare_checkpoints_notes(tmp_path):
    """Compare mock checkpoints and verify notes contain 'manual mapping'."""
    rllib_ckpt = tmp_path / "rllib_ckpt"
    rllib_ckpt.mkdir()
    (rllib_ckpt / "metadata.json").write_text('{"algorithm": "PPO"}')

    import zipfile, torch, io
    sb3_zip = tmp_path / "sb3_model.zip"
    state_dict = {"actor.weight": torch.randn(64, 10)}
    policy_bytes = io.BytesIO()
    torch.save(state_dict, policy_bytes)
    policy_bytes.seek(0)
    with zipfile.ZipFile(sb3_zip, "w") as z:
        z.writestr("policy.pth", policy_bytes.read())

    result = compare_checkpoints(str(rllib_ckpt), str(sb3_zip))
    assert "manual mapping" in result["notes"].lower()


def test_inspect_sb3_without_sb3_warns(tmp_path, monkeypatch):
    """SB3 not installed: gracefully warns."""
    import zipfile
    sb3_zip = tmp_path / "sb3_model.zip"
    with zipfile.ZipFile(sb3_zip, "w") as z:
        z.writestr("policy.pth", b"fake")
    # stable-baselines3 IS installed in dev, so this test is for coverage
    info = inspect_sb3_checkpoint(str(sb3_zip))
    assert info["format"] == "sb3"
```

```python
import subprocess
import sys


def test_distributed_extra_resolves():
    """Verify pip can resolve [distributed] extra without conflicts."""
    result = subprocess.run(
        [sys.executable, "-m", "pip", "install", "--dry-run", "-e", ".[distributed]"],
        capture_output=True,
        text=True,
        cwd=".",
    )
    assert result.returncode == 0, f"pip dry-run failed: {result.stderr}"
    assert "ray" in result.stdout.lower() or "ray" in result.stderr.lower()
```
