"""Tests for Ros2BridgeConfig — Pydantic v2 validation, defaults, YAML parsing."""

import tempfile
import pytest
from pydantic import ValidationError
from surg_rl.ros2.config import Ros2BridgeConfig


class TestRos2BridgeConfig:
    """Unit tests for Ros2BridgeConfig dataclass."""
    
    def test_defaults(self):
        """All optional fields have correct defaults."""
        cfg = Ros2BridgeConfig(state_topic="/surg_rl/joint_states", command_topic="/surg_rl/commands")
        assert cfg.frame_id == "world"
        assert cfg.batch_size == 1
        assert cfg.qos_profile == "sensor_data"
        assert cfg.on_missing_topic == "error"
        assert cfg.on_nan_inf == "raise"
        assert cfg.on_dimension_mismatch == "zero"
    
    def test_missing_required(self):
        """Missing required fields raise ValidationError."""
        with pytest.raises(ValidationError):
            Ros2BridgeConfig(state_topic="/a")  # command_topic missing
        with pytest.raises(ValidationError):
            Ros2BridgeConfig(command_topic="/b")  # state_topic missing
    
    def test_invalid_types(self):
        """Invalid field types raise ValidationError."""
        with pytest.raises(Exception):  # ValidationError or TypeError
            Ros2BridgeConfig(state_topic="/a", command_topic="/b", batch_size="abc")
        with pytest.raises(Exception):
            Ros2BridgeConfig(state_topic=123, command_topic="/b")
    
    def test_from_yaml_valid(self):
        """from_yaml parses valid YAML file correctly."""
        yaml_content = """state_topic: /custom/state
command_topic: /custom/cmd
frame_id: base_link
batch_size: 5
on_nan_inf: sanitize
on_dimension_mismatch: zero
"""
        with tempfile.NamedTemporaryFile(mode='w', suffix='.yaml', delete=False) as f:
            f.write(yaml_content)
            f.flush()
            
            cfg = Ros2BridgeConfig.from_yaml(f.name)
            assert cfg.state_topic == "/custom/state"
            assert cfg.command_topic == "/custom/cmd"
            assert cfg.frame_id == "base_link"
            assert cfg.batch_size == 5
            assert cfg.on_nan_inf == "sanitize"
    
    def test_from_yaml_missing_file(self):
        """from_yaml handles missing file gracefully."""
        cfg = Ros2BridgeConfig.from_yaml("/nonexistent/path.yaml")
        assert cfg.state_topic == "/surg_rl/joint_states"
        assert cfg.command_topic == "/surg_rl/commands"
    
    def test_from_yaml_malformed(self):
        """from_yaml handles malformed YAML with error."""
        yaml_content = "state_topic: [invalid\n  - missing_quote"
        with tempfile.NamedTemporaryFile(mode='w', suffix='.yaml', delete=False) as f:
            f.write(yaml_content)
            f.flush()
            with pytest.raises(Exception):
                Ros2BridgeConfig.from_yaml(f.name)
    
    def test_all_fields_custom(self):
        """All fields accept custom values."""
        cfg = Ros2BridgeConfig(
            state_topic="/custom/state",
            command_topic="/custom/cmd",
            frame_id="robot_base",
            batch_size=10,
            qos_profile="default",
            on_missing_topic="warn",
            on_nan_inf="sanitize",
            on_dimension_mismatch="zero",
        )
        assert cfg.state_topic == "/custom/state"
        assert cfg.command_topic == "/custom/cmd"
        assert cfg.frame_id == "robot_base"
        assert cfg.batch_size == 10
        assert cfg.qos_profile == "default"
        assert cfg.on_missing_topic == "warn"
        assert cfg.on_nan_inf == "sanitize"
        assert cfg.on_dimension_mismatch == "zero"

"""Tests for EnvironmentController ROS2 mode switching."""

import numpy as np
import queue
import pytest
from surg_rl.dynamics.environment_controller import (
    EnvironmentController,
    EnvironmentControllerConfig,
)


class TestControllerRos2Mode:
    """Tests for real/sim mode switch and external action injection."""
    
    def test_default_mode_is_sim(self):
        """New controller starts in sim mode."""
        controller = EnvironmentController()
        assert controller.mode == "sim"
    
    def test_set_real_robot_mode(self):
        """set_real_robot_mode switches mode correctly."""
        controller = EnvironmentController()
        controller.set_real_robot_mode(True)
        assert controller.mode == "real_robot"
        controller.set_real_robot_mode(False)
        assert controller.mode == "sim"
    
    def test_get_action_sim_passthrough(self):
        """In sim mode, get_action returns policy action unchanged."""
        controller = EnvironmentController()
        policy_action = np.array([1.0, 2.0, 3.0])
        result = controller.get_action(policy_action)
        np.testing.assert_array_equal(result, policy_action)
    
    def test_get_action_real_robot_external(self):
        """In real_robot mode, get_action returns injected external action."""
        controller = EnvironmentController()
        controller.set_real_robot_mode(True)
        
        external = np.array([4.0, 5.0, 6.0])
        controller.inject_external_action(external)
        
        result = controller.get_action(np.array([1.0, 2.0, 3.0]))
        np.testing.assert_array_equal(result, external)
    
    def test_get_action_real_robot_empty_queue(self):
        """Empty queue in real_robot mode returns last_action."""
        controller = EnvironmentController()
        controller.set_real_robot_mode(True)
        # No action injected yet — last_action is None
        # get_action returns hold-last (None → policy_action as fallback)
        policy = np.array([1.0, 0.0])
        result = controller.get_action(policy)
        # When _last_action is None and queue empty, returns policy
        # This is the initial-state behavior before first command
        assert result is not None
    
    def test_inject_keep_latest(self):
        """Third inject overwrites second in maxsize=1 queue."""
        controller = EnvironmentController()
        controller.set_real_robot_mode(True)
        
        controller.inject_external_action(np.array([1.0]))
        controller.inject_external_action(np.array([2.0]))
        controller.inject_external_action(np.array([3.0]))
        
        result = controller.get_action(np.array([0.0]))
        np.testing.assert_array_equal(result, np.array([3.0]))
    
    def test_get_status_includes_mode(self):
        """get_status dict includes mode field."""
        controller = EnvironmentController()
        status = controller.get_status()
        assert "mode" in status
        assert status["mode"] == "sim"
    
    def test_existing_reset_still_works(self):
        """Controller.reset() works normally after adding mode."""
        controller = EnvironmentController(
            config=EnvironmentControllerConfig(
                enabled=True,
                use_randomization=False,
            )
        )
        params = controller.reset(seed=42)
        assert params is not None
        assert params.episode == 1

"""Tests for TrajectoryReplay — speed validation, import guards."""

import pytest


class TestTrajectoryReplay:
    """Unit tests for trajectory replay logic."""
    
    def test_speed_validation(self):
        """Speed must be in (0.0, 1.0]."""
        from surg_rl.ros2.replay import TrajectoryReplay
        import sys
        
        if sys.platform == "darwin":
            # On macOS, __init__ raises RuntimeError before speed check
            with pytest.raises(RuntimeError):
                TrajectoryReplay("fake.zip", "fake.json", speed=0)
        else:
            # On Linux without rclpy: also RuntimeError
            # On Linux with rclpy: ValueError for bad speed
            try:
                TrajectoryReplay("fake.zip", "fake.json", speed=0)
            except RuntimeError as e:
                assert "ROS2 is not available" in str(e)
            except ValueError as e:
                assert "Speed" in str(e)
    
    def test_speed_negative(self):
        """Negative speed rejected."""
        from surg_rl.ros2.replay import TrajectoryReplay
        try:
            TrajectoryReplay("fake.zip", "fake.json", speed=-0.5)
        except (RuntimeError, ValueError):
            pass  # Either error is acceptable
    
    def test_speed_over_one(self):
        """Speed > 1.0 rejected."""
        from surg_rl.ros2.replay import TrajectoryReplay
        try:
            TrajectoryReplay("fake.zip", "fake.json", speed=1.5)
        except (RuntimeError, ValueError):
            pass  # Either error is acceptable
    
    def test_runtime_error_on_macos(self):
        """Clear RuntimeError with apt instructions on macOS."""
        from surg_rl.ros2.replay import TrajectoryReplay
        import sys
        try:
            TrajectoryReplay("fake.zip", "fake.json")
        except RuntimeError as e:
            assert "ROS2 is not available" in str(e)
            assert "apt" in str(e) or "rclpy" in str(e)

"""Tests for ROS2 CLI commands — help output, error cases."""

import pytest
from typer.testing import CliRunner
from surg_rl.cli import app


runner = CliRunner()


class TestRos2Cli:
    """CLI tests for ros2-bridge and ros2-replay commands."""
    
    def test_ros2_bridge_help(self):
        """ros2-bridge --help shows usage."""
        result = runner.invoke(app, ["ros2-bridge", "--help"])
        assert result.exit_code == 0
        assert "--config" in result.stdout
        assert "--scene" in result.stdout
    
    def test_ros2_replay_help(self):
        """ros2-replay --help shows usage."""
        result = runner.invoke(app, ["ros2-replay", "--help"])
        assert result.exit_code == 0
        assert "--checkpoint" in result.stdout
        assert "--speed" in result.stdout
    
    def test_ros2_bridge_missing_config(self):
        """Missing --config prints error."""
        result = runner.invoke(app, ["ros2-bridge"])
        # May exit 0 on macOS (early guard), 1 on Linux without config
        assert result.exit_code in (0, 1)
    
    def test_ros2_replay_missing_checkpoint(self):
        """Missing --checkpoint fails."""
        result = runner.invoke(app, ["ros2-replay"])
        assert result.exit_code != 0
    
    def test_cli_commands_registered(self):
        """ros2-bridge and ros2-replay appear in app commands."""
        command_names = [c.name for c in app.registered_commands]
        assert "ros2-bridge" in command_names
        assert "ros2-replay" in command_names
