```python
class Ros2Bridge:
    """Manages Ros2BridgeNode lifecycle as a separate multiprocessing Process."""
    
    def __init__(self, node, config: "Ros2BridgeConfig"):
        self._node = node
        self._config = config
        self._process: "multiprocessing.Process | None" = None
        self._running = False
    
    def start(self) -> None:
        """Spawn bridge process."""
        import multiprocessing
        self._process = multiprocessing.Process(
            target=self._run_bridge,
            daemon=True,  # dies with parent
            name="ros2-bridge",
        )
        self._process.start()
        self._running = True
    
    def terminate(self) -> None:
        """Terminate bridge process and join."""
        if self._process is not None and self._process.is_alive():
            self._process.terminate()
            self._process.join(timeout=5.0)
            if self._process.is_alive():
                self._process.kill()
                self._process.join(timeout=2.0)
            self._running = False
    
    def publish_joint_state(self, qpos: "np.ndarray", qvel: "np.ndarray") -> None:
        """Publish joint state from main process — delegates to node."""
        if self._running:
            self._node.publish_state(qpos, qvel)
    
    @staticmethod
    def _run_bridge() -> None:
        """Bridge process main function. Initializes rclpy and spins."""
        import rclpy
        from rclpy.executors import MultiThreadedExecutor
        rclpy.init()
        executor = MultiThreadedExecutor()
        executor.add_node(_bridge_node_instance)
        try:
            executor.spin()
        except KeyboardInterrupt:
            pass
        finally:
            executor.shutdown()
            _bridge_node_instance.destroy_node()
            rclpy.shutdown()
```

Note on multiprocessing.Process pattern for node: Since `Ros2BridgeNode` methods need to be called from the main process (`publish_state`) but `rclpy.spin()` runs in the child process, the node reference is shared via a module-level variable or the bridge wrapper stores it. Simplest approach per RESEARCH.md: the main thread calls `node.publish_state()` directly on the node object (rclpy Node is thread-safe for publish), and the child process spins with `rclpy.spin()`. Store node in a module-level or bridge-wrapper accessible variable.
