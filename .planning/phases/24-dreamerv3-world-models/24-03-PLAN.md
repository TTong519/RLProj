```python
def _run_dreamer_evaluation(self, task: str, backend: str) -> dict:
    """Run DreamerV3 evaluation for a task/backend if checkpoint exists."""
    from surg_rl.dreamer.training import evaluate_checkpoint
    from pathlib import Path
    
    # Auto-discover checkpoint from standard location
    for obs_type in ["pixels", "state"]:
        checkpoint_dir = Path(f"models/dreamerv3/{task}_{obs_type}")
        if checkpoint_dir.exists():
            checkpoints = list(checkpoint_dir.glob("checkpoint_*.pt"))
            if checkpoints:
                latest = max(checkpoints, key=lambda p: p.stat().st_mtime)
                # DreamerV3 eval doesn't care about backend (runs in subprocess)
                # But we report per backend for consistency
                metrics = evaluate_checkpoint(str(latest), task, obs_type, self.config.eval_episodes)
                metrics["obs_type"] = obs_type
                metrics["checkpoint"] = str(latest)
                return metrics
    return {"status": "pending — Phase 24", "reason": "no checkpoint found"}
```

     - Include obs_type in metadata
