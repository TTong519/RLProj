<objective>
Make simulator state save/restore behaviorally equivalent across MuJoCo and PyBullet backends, including soft-body node data. Make TrainingManager.evaluate() reuse existing vectorized environments instead of creating fresh instances per call.

Purpose: RL training requires deterministic episode resets and fast evaluation. Current PyBullet set_state() does not restore joint states or soft-body deformations, causing state divergence after get→set cycles. TrainingManager.evaluate() creates a new env every call, wasting seconds per evaluation interval.
Output: Full round-trip state fidelity in both backends with tests, and persistent eval env caching in TrainingManager.
</objective>

def get_state(self) -> State:
    body_positions = {}
    body_orientations = {}
    for name, body_id in self._body_ids.items():
        pos, orn = self._pb.getBasePositionAndOrientation(body_id, ...)
        body_positions[name] = np.array(pos)
        body_orientations[name] = np.array(orn)
    joint_states = self.get_joint_states()
    qpos_list = []
    qvel_list = []
    for _robot_name, states in joint_states.items():
        qpos_list.append(states["positions"])
        qvel_list.append(states["velocities"])
    qpos = np.concatenate(qpos_list) if qpos_list else None
    qvel = np.concatenate(qvel_list) if qvel_list else None
    return State(time=self._simulation_time, qpos=qpos, qvel=qvel,
                 body_positions=body_positions, body_orientations=body_orientations)

def set_state(self, state: State) -> None:
    self._simulation_time = state.time
    for name, pos in state.body_positions.items():
        if name in self._body_ids:
            orn = state.body_orientations.get(name, [0, 0, 0, 1])
            self._pb.resetBasePositionAndOrientation(
                self._body_ids[name], pos.tolist(), orn.tolist() if hasattr(orn, "tolist") else orn, ...
            )
```

def get_state(self) -> State:
    body_positions = {}
    body_orientations = {}
    if self._scene:
        for robot in self._scene.robots:
            pose = self.get_body_pose(robot.name)
            if pose is not None:
                body_positions[robot.name] = pose[0]
                body_orientations[robot.name] = pose[1]
        for tissue in self._scene.tissues:
            pose = self.get_body_pose(tissue.name)
            if pose is not None:
                body_positions[tissue.name] = pose[0]
                body_orientations[tissue.name] = pose[1]
    return State(time=self._simulation_time,
                 qpos=self._data.qpos.copy() if self._data.qpos is not None else None,
                 qvel=self._data.qvel.copy() if self._data.qvel is not None else None,
                 mocap_pos=self._data.mocap_pos.copy() if hasattr(self._data, "mocap_pos") else None,
                 mocap_quat=self._data.mocap_quat.copy() if hasattr(self._data, "mocap_quat") else None,
                 body_positions=body_positions, body_orientations=body_orientations)

def set_state(self, state: State) -> None:
    self._simulation_time = state.time
    if state.qpos is not None:
        self._data.qpos[:] = state.qpos
    if state.qvel is not None:
        self._data.qvel[:] = state.qvel
    if state.mocap_pos is not None and hasattr(self._data, "mocap_pos"):
        self._data.mocap_pos[:] = state.mocap_pos
    if state.mocap_quat is not None and hasattr(self._data, "mocap_quat"):
        self._data.mocap_quat[:] = state.mocap_quat
    self._mujoco.mj_forward(self._model, self._data)
```
