```python
"""K8S-01..05: Validate Kubernetes manifests via dry-run and structure checks."""
from __future__ import annotations

import subprocess
import sys
from pathlib import Path

import pytest

K8S_BASE = Path(__file__).resolve().parents[1] / "k8s" / "base"

HAS_KUBECTL = subprocess.run(["which", "kubectl"], capture_output=True).returncode == 0

MANIFESTS = ["training-job.yaml", "raycluster.yaml", "rayjob.yaml", "configmap.yaml", "secret.yaml", "pvc.yaml", "rbac.yaml"]


def _dry_run(path: Path) -> bool:
    if not HAS_KUBECTL:
        return True
    r = subprocess.run(
        ["kubectl", "apply", "-f", str(path), "--dry-run=client"],
        capture_output=True, text=True,
    )
    return r.returncode == 0


class TestManifestSyntax:
    @pytest.mark.parametrize("manifest", MANIFESTS)
    def test_dry_run_passes(self, manifest):
        assert _dry_run(K8S_BASE / manifest), f"{manifest} dry-run failed"

    def test_cpu_overlay_passes(self):
        r = subprocess.run(
            ["kubectl", "apply", "-k", str(K8S_BASE.parent / "overlays" / "cpu"), "--dry-run=client"],
            capture_output=True, text=True,
        )
        assert r.returncode == 0, r.stderr

    def test_gpu_overlay_passes(self):
        r = subprocess.run(
            ["kubectl", "apply", "-k", str(K8S_BASE.parent / "overlays" / "gpu"), "--dry-run=client"],
            capture_output=True, text=True,
        )
        assert r.returncode == 0, r.stderr


class TestJobStructure:
    def test_has_gpu_node_selector(self):
        import yaml
        job = yaml.safe_load((K8S_BASE / "training-job.yaml").read_text())
        sel = job["spec"]["template"]["spec"]["nodeSelector"]
        assert "nvidia.com/gpu.present" in sel

    def test_has_resource_limits(self):
        import yaml
        job = yaml.safe_load((K8S_BASE / "training-job.yaml").read_text())
        limits = job["spec"]["template"]["spec"]["containers"][0]["resources"]["limits"]
        assert "cpu" in limits
        assert "memory" in limits

    def test_has_pvc_volume(self):
        import yaml
        job = yaml.safe_load((K8S_BASE / "training-job.yaml").read_text())
        volumes = {v["name"]: v for v in job["spec"]["template"]["spec"]["volumes"]}
        assert "checkpoints" in volumes
        assert "persistentVolumeClaim" in volumes["checkpoints"]
```

```python
"""K8S-02: Verify train_rllib respects RAY_ADDRESS env var."""
from __future__ import annotations

import os
from unittest.mock import MagicMock, patch


def test_ray_address_env_var_respected():
    import surg_rl.rl.rllib.train as train_mod
    with patch.object(train_mod, "ray") as mock_ray:
        mock_ray.is_initialized.return_value = False
        os.environ["RAY_ADDRESS"] = "ray://head-svc:10001"
        train_mod.train_rllib(train_mod.RllibConfig(), stop_criteria={"num_env_steps_sampled_lifetime": 0})
        assert mock_ray.init.called
        kwargs = mock_ray.init.call_args.kwargs
        assert kwargs.get("address") == "ray://head-svc:10001"

def test_ray_address_defaults_to_auto():
    import surg_rl.rl.rllib.train as train_mod
    with patch.object(train_mod, "ray") as mock_ray:
        mock_ray.is_initialized.return_value = False
        os.environ.pop("RAY_ADDRESS", None)
        train_mod.train_rllib(train_mod.RllibConfig(), stop_criteria={"num_env_steps_sampled_lifetime": 0})
        assert mock_ray.init.called
        kwargs = mock_ray.init.call_args.kwargs
        assert kwargs.get("address") == "auto"
```
