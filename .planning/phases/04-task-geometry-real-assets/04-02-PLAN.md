    def __init__(self, assets_dir=None, use_primitive_fallback=True):
        self.assets_dir = Path(assets_dir) if assets_dir else None
        self.use_primitive_fallback = use_primitive_fallback
        self._temp_dir_obj = tempfile.TemporaryDirectory(prefix="surg_rl_")
        self.temp_dir = Path(self._temp_dir_obj.name)
        self._primitive_meshes: dict[str, Path] = {}
        self._vtk_meshes: dict[str, Path] = {}

    def resolve_asset_path(self, asset_path: str) -> Path | None:
        path = Path(asset_path)
        if path.is_absolute() and path.exists():
            return path
        if self.assets_dir:
            full_path = self.assets_dir / path
            if full_path.exists():
                return full_path
        if path.exists():
            return path.resolve()
        return None

## AGENTS.md note
"Scene assets (URDFs / meshes) do not exist in `assets/`. `scene_builder` generates primitive `.obj` fallbacks on the fly. Never assume a real asset file exists."

- OBJ: visual mesh only (collision geometry still from primitives)
- Missing asset: single `logger.warning` at `load_scene()` time, `self._missing_assets: set[str]` prevents duplicates
