```python
def _step_simulator_and_build_outputs(
    self, processed_action: np.ndarray
) -> tuple[dict[str, np.ndarray], float, bool, bool, dict[str, Any]]:
    """Build (obs, reward, terminated, truncated, info) given an already-applied action.

    Shared body used by both step() and passthrough_step(). Does NOT re-apply
    the action to the simulator — caller must do that first.
    """
    # (verbatim body from current step() lines 572-663, minus the simulator.step() call)
    try:
        step_result = self._simulator.step(processed_action)
        sim_obs = step_result.observation
        reward = step_result.reward
        terminated = step_result.terminated
        truncated = step_result.truncated
        sim_info = step_result.info
    except Exception as e:
        logger.warning(f"Simulator step failed: {e}")
        from surg_rl.simulators.base_simulator import Observation
        sim_obs = Observation()
        reward = 0.0
        terminated = True
        truncated = False
        sim_info = {"error": str(e)}
    # ... rest of body (step_count, truncation check, obs_builder, reward, success, info, fluid, bridge, controller) ...
    return obs_dict, reward, terminated, truncated, info

def passthrough_step(self) -> tuple[dict[str, np.ndarray], float, bool, bool, dict[str, Any]]:
    """Step environment when per-arm actions were applied via simulator.apply_action(arm_id=...) directly.

    MultiAgentSurgicalEnv routes per-arm actions to the simulator BEFORE calling
    this method, so the surgical env's action builder is bypassed entirely.

    Returns:
        Tuple of (observation, reward, terminated, truncated, info).
    """
    if self._simulator is None:
        raise RuntimeError("SurgicalEnv.passthrough_step() called before simulator loaded")
    num_controls = self._simulator.get_num_controls()
    no_op = np.zeros(num_controls, dtype=np.float32)
    return self._step_simulator_and_build_outputs(no_op)
```

**Do NOT** add the per-arm action application to `passthrough_step()` itself — that responsibility stays with `MultiAgentSurgicalEnv.step()` per D-11 (passthrough composition).
