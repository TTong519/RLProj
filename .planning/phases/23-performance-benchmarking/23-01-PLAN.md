```python
class LazyImport:
    """Deferred import that raises ImportError on first attribute access."""
    
    def __init__(self, module_name: str, extra_name: str):
        self._module_name = module_name
        self._extra_name = extra_name
        self._module = None
        self._error = None
    
    @property
    def available(self) -> bool:
        """Check if the optional dependency is available without triggering import."""
        if self._module is not None:
            return True
        if self._error is not None:
            return False
        try:
            import importlib
            importlib.util.find_spec(self._module_name)
            return True
        except Exception:
            return False

    def __getattr__(self, name: str):
        if self._module is None:
            self._load()
        return getattr(self._module, name)
    
    def _load(self):
        try:
            import importlib
            self._module = importlib.import_module(self._module_name)
        except ImportError as e:
            self._error = e
            raise ImportError(
                f"Optional dependency '{self._module_name}' not found. "
                f"Install with: pip install surg-rl[{self._extra_name}]"
            ) from e
```
