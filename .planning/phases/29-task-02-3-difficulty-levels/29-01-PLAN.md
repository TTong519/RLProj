    from surg_rl.rl import DifficultyLevel

        (SuturingReward, ["needle_position_tolerance", "thread_tension_threshold", "stitch_spacing_tolerance", "time_limit"], []),

        (GraspingReward, ["approach_tolerance", "grip_force_accuracy", "time_limit"], ["object_mass", "action_noise"]),
        (CuttingReward, ["cut_path_accuracy", "collateral_threshold", "force_precision", "time_limit"], ["tissue_stiffness"]),
    ])
    def test_difficulty_direction(reward_cls, down_keys, up_keys):
        """D-DIR-01: per-family direction assertion between EASY and HARD."""
        easy = reward_cls.get_params_for_difficulty(DifficultyLevel.EASY)
        hard = reward_cls.get_params_for_difficulty(DifficultyLevel.HARD)
        for name in down_keys:
            assert hard[name] < easy[name], (
                f"{reward_cls.__name__}: {name} did not move strict (HARD<{name}<EASY). "
                f"Easy={easy[name]}, Hard={hard[name]}"
            )
        for name in up_keys:
            assert hard[name] > easy[name], (
                f"{reward_cls.__name__}: {name} did not move strict (HARD>{name}>EASY). "
                f"Easy={easy[name]}, Hard={hard[name]}"
            )

    # the ctor field that `apply_difficulty` mutates; we assert the value
    # changes after calling apply_difficulty(HARD).
    MAPPED_FIELDS: dict[type, str] = {
        SuturingReward: "position_threshold",       # needle_position_tolerance → 0.002
        DissectionReward: "force_threshold",        # force_precision → 1.0
        NeedlePassingReward: "handoff_threshold",   # handoff_proximity_tolerance → 0.01
        KnotTyingReward: "loop_deviation_threshold",# loop_deviation_tolerance → 0.005
        GraspingReward: "grasp_threshold",          # approach_tolerance → 0.005
        CuttingReward: "force_threshold",           # force_precision → 1.0

        field_name = MAPPED_FIELDS[reward_cls]
